Get a list of in-service dates for aircraft carrier

# https://www.wikidata.org/wiki/Wikidata:SPARQL_tutorial#Qualifiers
WikidataQueryServiceR::query_wikidata(
  "
  SELECT ?ship ?shipLabel ?event ?eventLabel ?time
WHERE
{
  ?ship wdt:P31/wdt:P279* wd:Q17205; # ship is a instance or subclass of aircraft carrier
            p:P793 [ ps:P793 wd:Q14475832; pq:P585 ?time ]. 
  VALUES ?event{wd:Q14475832}
  SERVICE wikibase:label { bd:serviceParam wikibase:language '[AUTO_LANGUAGE]'. }
}
  "
)
